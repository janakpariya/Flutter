import 'package:audioplayers/audioplayers.dart';
import 'package:flutter/material.dart';

void main() => runApp(Home());

class Home extends StatelessWidget {
  const Home({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: SongApp(),
    );
  }
}

class SongApp extends StatefulWidget {
  const SongApp({Key? key}) : super(key: key);
  @override
  State<SongApp> createState() => _SongAppState();
}

class _SongAppState extends State<SongApp> {

  AudioPlayer audioPlayer = AudioPlayer();
  String myAudioPath = "audio/faded.mp3";

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Play Song'),
      ),

      body: Container(
        child: ListView(

          children: [

            Container(
              alignment: Alignment.center,
              margin: EdgeInsets.all(30.0),
              child: Row(
                children: [

                  Expanded(
                    child: IconButton(
                      onPressed: () async {
                        await audioPlayer.play(AssetSource(myAudioPath));
                      },
                      icon: Icon(Icons.play_arrow),
                    ),
                  ),

                  SizedBox(
                    width: 10.0,
                  ),

                  Expanded(
                    child: IconButton(
                      onPressed: () async {
                        await audioPlayer.resume();
                      },
                      icon: Icon(Icons.restart_alt),
                    ),
                  ),

                  SizedBox(
                    width: 10.0,
                  ),

                  Expanded(
                    child: IconButton(
                      onPressed: () async {
                        await audioPlayer.pause();
                      },
                      icon: Icon(Icons.pause),
                    ),
                  ),

                  SizedBox(
                    width: 10.0,
                  ),

                  Expanded(
                    child: IconButton(
                      onPressed: () async {
                        await audioPlayer.stop();
                      },
                      icon: Icon(Icons.stop),
                    ),
                  ),

                ],
              ),

            ),
          ],
        ),
      ),
    );
  }
}
